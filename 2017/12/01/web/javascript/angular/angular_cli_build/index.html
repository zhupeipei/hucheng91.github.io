  2. 小明和小芳协同开发某个业务，小明负责后端接口，小芳负责前端页面，现在小芳想本地接口调试，小明把接口发布到远端dev服务器，小芳需要本地连接dev环境接口
  3. 小芳是个前端工程师，qa环境出现了bug，本地想重现这个bug

// 拦截所有 /MidNodeJS开头的的ajax请求
const PROXY_CONFIG = {
“/MidNodeJS”: {
“target”: “http://localhost:8080/“,
“secure”: false,
“changeOrigin”: true,
“logLevel”: “debug”,
“pathRewrite”: { // 这个参数是选的,意思是/MidNodeJS/转换成/
“^/MidNodeJS”: “”
},
“bypass”: function (req, res, proxyOptions) {
//这里面是伪造浏览器的个各种参数
req.headers[“x-app-code”] = “race”;
req.headers[“referer”] = “https://raceqa.saic-gm.com/“;
}
}
}
module.exports = PROXY_CONFIG;

const PROXY_CONFIG = {
“/MidNodeJS”: {
“target”: “http://10.203.102.139:80/“,
“secure”: false,
“changeOrigin”: true,
“logLevel”: “debug”,
“pathRewrite”: {
“^/MidNodeJS”: “”
},
“bypass”: function (req, res, proxyOptions) {

  req.headers["x-app-code"] = "race";
  req.headers["referer"] = "https://raceqa.saic-gm.com/";
}

}
}

module.exports = PROXY_CONFIG;
